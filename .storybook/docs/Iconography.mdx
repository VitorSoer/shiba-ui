import { t } from 'i18next';
import * as Icon from '../../src/assets/icons';
import {DocIconWrapper} from '../../src/components/Feedback/Icon/styles';
import { Meta, Title, IconGallery, IconItem } from '@storybook/blocks';
import { DocsTitle, DocsDescription } from '../i18n/components/TextElements';

<Meta title="Design System/Icons" />

<DocsTitle text="icons_title" />
<DocsDescription text="icons_description" />

<IconGallery>
  {Object.entries(Icon).map(([name, icon]) => {
    const handleClick = () => {
      alert(t('docs_copied'))
      navigator.clipboard.writeText(name)
    }

    return (
      <DocIconWrapper key={name} onClick={handleClick}>
        <svg width="24" height="24" viewBox="0 0 24 24" fill="none">
          <path d={icon.path.join(' ')} fill="currentColor" />
        </svg>

        <p>{name}</p>
      </DocIconWrapper>
  )})}
</IconGallery>
